在进入index之后，有三种情况，分别是

  1. 从登录页面进来的用户，会显示教程提示，远程刷新
  2. 之前登录过的老用户，localStorage还存着用户名信息，本地刷新
  3. 没登录的用户，直接跳转到登录界面（没有账号可以再登录界面前往注册界面）
window.onload = function () {

2. 本地存储与云端延时同步

本地的即时修改存储在localStorage中，并不是所有对todo的修改都会进行一次服务器的同步，保证了响应时间的高效，在进行本地修改（修改文字，修改完成状态、删除todo）的时候，会触发一次3秒的等待，若这60s内没有其他修改，则将本地的todo修改提交到服务器，若这60s还有其他修改，则秒数重置。这样做的好处是，如果在一段时间内有频繁的修改，那么只会与服务器交换一次信息，节约响应时间。（很多地方都考虑了一些微小的用户体验）

也可以点击右上角第二个同步按钮，直接保存到服务器：

相关的代码：（物理位置不在一起，这里只是贴在了一起）

var syncServer = 0;

// 本地localStorage改动后触发此函数，当用户3秒之内无其他改动后，会同步到云端
function prepareSync() {
    syncServer += 1;
    console.log(syncServer);
    setTimeout(function () {
        syncServer -= 1;
        if (syncServer === 0) {
            SyncServer();
        }
    }, 3000)
}

// 将本地localStorage数据同步到云端
function SyncServer() {
    console.log("自动保存");
    showMsg("自动保存中...");
    uploadTodo();
    //setTimeout(flushFromServer, 2000);
}

function uploadTodo() {
    var todos;
    if (localStorage.getItem("todo")) {
        todos = JSON.parse(localStorage.getItem("todo"))
    } else {
        todos = [];
    }
    var newTodo = JSON.stringify({todo: JSON.stringify(todos)});
    var username = localStorage.getItem("username");
    let xmlHttp = new XMLHttpRequest();
    xmlHttp.onreadystatechange = function () {
        if (xmlHttp.readyState === 4 && xmlHttp.status === 200) {
            console.log(xmlHttp.responseText);
        }
    };
    xmlHttp.open("post", "https://yicheng.me/todoapi/updateTodo/" + username.toString());
    xmlHttp.setRequestHeader("content-type", "application/json");
    xmlHttp.send(newTodo);
}

4. 用户注册

注册界面如下：

既然都部署了前后端，那正常的用户注册登录肯定是少不了的，用户注册会有账号空缺提示、密码空缺提示与用户名重复提示，发送这些情况不会注册成功。

主要的js代码如下（常规的ajax请求）

// 注册表单提交
function submitRegister()
{
    let username = document.getElementsByName("username")[0].value;
    let password = document.getElementsByName("password")[0].value;
    if (username === "") {
        showMsg("用户名不能为空！");
    } else if (password === "") {
        showMsg("密码不能为空！");
    } else {
        var formData = JSON.stringify({"name": username.toString(), "password": password.toString()});
        let xmlHttp = new XMLHttpRequest();
        console.log(formData);
        xmlHttp.onreadystatechange = function()
        {
            if(xmlHttp.readyState === 4 && xmlHttp.status === 200)
            {
                console.log(xmlHttp.responseText);
                showMsg("注册成功！即将跳转到登录页面");
                setTimeout(function () {
                    window.location.replace("login.html");
                }, 2000);
            } else if(xmlHttp.status === 204) {
                showMsg("这个用户名已经存在了，请换一个！");
            }
        };
        xmlHttp.open("post", "https://yicheng.me/todoapi/register");
        xmlHttp.setRequestHeader("content-type","application/json");
        xmlHttp.send(formData);
    }
}

当然UI缺点也很明显，按钮太多了，没有手机端常见的滑动、长按等交互，等老师讲到手机适配那节课，我才意识到这个问题，然而基本都写好了，就没有加上，以后有空再增强一下移动端交互有关的操作。

19. 其他一些小功能点

其他的一些小功能点其实也是有的，但是比较小，就不值得单独算一点说了，在这里一起说一下

19.1 点击左上角用户头像，进入用户详情页，不过该页面比较简陋，可以注销账号，回到登录页面
19.2 todo的高度有限制，超过之后右侧会显示滚动条（滚动条的颜色也可以随着主题改变切换颜色）
19.3 一些hover悬浮，虽然老师也提过，在移动端表现不如PC端，但还是能看的
